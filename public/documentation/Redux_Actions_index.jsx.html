/**
 * Method to show the todos
 This one always whows todo in the case happens the following issuse:
  1. User filters the todos
  2. User searches todos
  3. User deletes todo
  4. User Edits todo
  5. User stopped sorting todos
 @this redux-action-showToDos
 @global
*/
export const showToDos = () => ({
	type: 'SHOW_TODO'
})

/**

  3. Priority

 @global
*/

	type: 'ADD_TO_DO',
